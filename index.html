//strict regime
    "use strict"
    
    //pulling out the weather block
    const weatherBlock = document.querySelector('#weather');
    
    //we make async function in order for the function 
    //to work correctly
    
    async function loadWeather(e){
    
    
    //we make a changeable const and assign it an api
    
      const server = 'https://api.openweathermap.org/data/2.5/weather?units=metric&q=Kyiv&appid=your_api_kay';
    
    //response we assign a fetch request by the get method to get data from the api
    
      const response = await fetch(server, {
    method:'GET',
    });
    
    //
    we receive information from the api in json format
    
    const responseResoult = await response.json();
    
    //we do a routine health check
    if(response.ok){
      getWeather(responseResoult);
    } else{
      weatherBlock.innerHTML = responseResoult.message;
    }
    
    function getWeather(data){
      console.log(data);
    //in the changeable ones, we assign parameters from the api for convenient use.
    //all information can be viewed in the console
    
      const location = data.name;
      const temp = Math.round(data. main.temp);
      const feelsLike = Math.round(data.main.feels_like);
      const weatherStatus = data.weather[0].main;
      const weatherIcon = data.weather[0].icon;
    
    //we are making a template structure for our html, this is so that //our weather is dynamic and does not look like shit 
    
      const template =`<div class="weather_header">
      <div class="weather_main">
        <div class="weather_city">${location}</div>
        <div class="weather_status">${weatherStatus}</div>
      </div>
      <div class="weather_icon">
    //here we upload a dynamic image from the site
        <img src="https://openweathermap.org/img/w/${weatherIcon}.png" alt="clouds">
      </div>
      <div class="weather_temp">${temp}</div>
      <div class="weather_feels-like">Feels like: ${feelsLike}</div>
    </div>    `;
    
    
    //we call our template and paste it into html
    
    weatherBlock.innerHTML=template;
    };
    
    }
    //calling the entire function using if
    if (weatherBlock) {
      loadWeather();
    }

the api key can be taken from the website https://openweathermap.org/api but you need to wait 24 hours after registration, and if you do not wait, the api will not work
